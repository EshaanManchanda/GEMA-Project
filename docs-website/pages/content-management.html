Category Model Structure


interface Category {
  id: string;
  name: string;
  slug: string;
  description?: string;
  parentId?: string;
  children?: Category[];
  metadata: {
    eventCount: number;
    isActive: boolean;
    displayOrder: number;
    icon?: string;
    color?: string;
  };
  timestamps: {
    createdAt: Date;
    updatedAt: Date;
  };
}

// Category management operations
class CategoryService {
  async createCategory(data: CreateCategoryInput): Promise<Category> {
    const category = await Category.create({
      ...data,
      slug: generateSlug(data.name),
      metadata: {
        eventCount: 0,
        isActive: true,
        displayOrder: 0,
        ...data.metadata
      }
    });
    
    return category;
  }

  async updateCategoryHierarchy(categoryId: string, newParentId?: string): Promise<void> {
    // Handle hierarchy updates with validation
    await this.validateHierarchy(categoryId, newParentId);
    await Category.update({ parentId: newParentId }, { where: { id: categoryId } });
  }
}

Cloudinary Integration


import { v2 as cloudinary } from 'cloudinary';

// Cloudinary configuration
cloudinary.config({
  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,
});

export class MediaService {
  async uploadImage(file: Buffer, options: UploadOptions): Promise<MediaResult> {
    try {
      const result = await cloudinary.uploader.upload_stream(
        {
          folder: options.folder || 'gema-events',
          transformation: [
            { width: 1200, height: 800, crop: 'fill', quality: 'auto' },
            { format: 'auto' }
          ],
          tags: options.tags || []
        },
        (error, result) => {
          if (error) throw error;
          return result;
        }
      );

      return {
        publicId: result.public_id,
        url: result.secure_url,
        width: result.width,
        height: result.height,
        format: result.format,
        bytes: result.bytes
      };
    } catch (error) {
      throw new MediaUploadError('Failed to upload media', error);
    }
  }

  generateOptimizedUrl(publicId: string, transformations: TransformOptions): string {
    return cloudinary.url(publicId, {
      ...transformations,
      secure: true,
      quality: 'auto',
      format: 'auto'
    });
  }
}
