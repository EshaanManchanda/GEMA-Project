Payment Gateway Integration


interface PaymentGateway {
  name: string;
  processPayment(request: PaymentRequest): Promise<PaymentResult>;
  refundPayment(paymentId: string, amount: number): Promise<RefundResult>;
  getPaymentStatus(paymentId: string): Promise<PaymentStatus>;
}

class StripeGateway implements PaymentGateway {
  name = 'stripe';
  
  async processPayment(request: PaymentRequest): Promise<PaymentResult> {
    const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);
    
    try {
      const paymentIntent = await stripe.paymentIntents.create({
        amount: request.amount,
        currency: request.currency,
        payment_method: request.paymentMethodId,
        confirm: true,
        metadata: {
          bookingId: request.bookingId,
          customerId: request.customerId,
          eventId: request.eventId
        }
      });
      
      return {
        success: true,
        transactionId: paymentIntent.id,
        status: paymentIntent.status,
        amount: paymentIntent.amount,
        fees: paymentIntent.charges.data[0]?.balance_transaction?.fee || 0
      };
    } catch (error) {
      return {
        success: false,
        error: error.message,
        errorCode: error.code
      };
    }
  }
}

class PaymentRouter {
  private gateways: Map<string, PaymentGateway> = new Map();
  
  constructor() {
    this.gateways.set('stripe', new StripeGateway());
    this.gateways.set('paypal', new PayPalGateway());
  }
  
  async processPayment(request: PaymentRequest): Promise<PaymentResult> {
    const preferredGateway = this.selectGateway(request);
    
    let result = await preferredGateway.processPayment(request);
    
    // Retry with fallback gateway if primary fails
    if (!result.success && this.shouldRetry(result.errorCode)) {
      const fallbackGateway = this.getFallbackGateway(preferredGateway.name);
      if (fallbackGateway) {
        result = await fallbackGateway.processPayment(request);
      }
    }
    
    // Log transaction for analytics
    await this.logTransaction(request, result, preferredGateway.name);
    
    return result;
  }
}

Commission Calculation Engine


interface CommissionRule {
  id: string;
  type: 'percentage' | 'fixed' | 'tiered' | 'performance';
  recipient: 'platform' | 'vendor' | 'employee' | 'affiliate';
  configuration: {
    rate?: number;
    amount?: number;
    tiers?: CommissionTier[];
    conditions?: CommissionCondition[];
  };
  priority: number;
  active: boolean;
}

class CommissionCalculator {
  async calculateCommissions(booking: Booking): Promise<CommissionBreakdown> {
    const breakdown: CommissionBreakdown = {
      grossAmount: booking.totalAmount,
      paymentFees: 0,
      commissions: [],
      netPayout: 0
    };
    
    // Calculate payment gateway fees
    breakdown.paymentFees = await this.calculatePaymentFees(booking);
    
    // Get applicable commission rules
    const rules = await this.getApplicableRules(booking);
    
    // Calculate each commission
    for (const rule of rules.sort((a, b) => a.priority - b.priority)) {
      const commission = await this.calculateRuleCommission(booking, rule, breakdown);
      breakdown.commissions.push(commission);
    }
    
    // Calculate final vendor payout
    const totalDeductions = breakdown.paymentFees + 
      breakdown.commissions.reduce((sum, c) => sum + c.amount, 0);
    breakdown.netPayout = breakdown.grossAmount - totalDeductions;
    
    return breakdown;
  }
  
  private async calculateRuleCommission(
    booking: Booking, 
    rule: CommissionRule, 
    currentBreakdown: CommissionBreakdown
  ): Promise<Commission> {
    const baseAmount = currentBreakdown.grossAmount - currentBreakdown.paymentFees;
    
    switch (rule.type) {
      case 'percentage':
        return {
          ruleId: rule.id,
          recipient: rule.recipient,
          type: 'percentage',
          rate: rule.configuration.rate,
          amount: baseAmount * (rule.configuration.rate / 100),
          description: `${rule.configuration.rate}% commission`
        };
        
      case 'tiered':
        const tier = this.findApplicableTier(booking, rule.configuration.tiers);
        return {
          ruleId: rule.id,
          recipient: rule.recipient,
          type: 'tiered',
          rate: tier.rate,
          amount: baseAmount * (tier.rate / 100),
          description: `Tier ${tier.level} - ${tier.rate}% commission`
        };
        
      case 'fixed':
        return {
          ruleId: rule.id,
          recipient: rule.recipient,
          type: 'fixed',
          amount: rule.configuration.amount,
          description: `Fixed commission of $${rule.configuration.amount}`
        };
        
      default:
        throw new Error(`Unsupported commission type: ${rule.type}`);
    }
  }
}
