Index Creation Script Example

// Create performance indexes for Users collection
db.users.createIndex({ email: 1 }, { unique: true });
db.users.createIndex({ role: 1, status: 1, createdAt: -1 });
db.users.createIndex({ "qrCode.code": 1 }, { unique: true, sparse: true });

// Create geospatial index for Events
db.events.createIndex({ "location.coordinates": "2dsphere" });

// Create text search index for Events
db.events.createIndex({
  title: "text",
  description: "text", 
  tags: "text"
}, {
  weights: {
    title: 10,
    description: 5,
    tags: 6
  }
});
